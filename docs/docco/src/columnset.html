    import { Column } from "./column";

export class ColumnSet {
    private keys: string[];
    private columns: Column[];
    private indexes: any;

    public constructor() {
        this.keys = [];
        this.columns = [];
        this.indexes = {};
    }

    public addColumn(key: string, data: string[]) {
        const col = new Column(key, data);
        this.columns.push(col);
        this.indexes[key] = col;
        return this;
    }

    public getColumn(key: string): Column | null {
        return (this.hasColumn(key) ? this.columns[this.indexes[key]] : null);
    }

    public getColumns(): Column[] {
        return this.columns;
    }

    public getColumnKeys(): string[] {
        return this.keys;
    }

    public hasColumn(key: string): boolean {
        return this.indexes.hasOwnProperty(key);
    }
}
